import matplotlib.pyplot as plt
import numpy as np


x = np.arange(0, 20, 0.2)
y1 = np.sin(x)
y2 = np.cos(x)

fig = plt.figure()
ax = fig.add_subplot(111)

ax.plot(x, y1, label="sin")
ax.plot(x, y2, label="cos")

ax.legend()
plt.show()

fig = plt.figure()
ax = fig.add_subplot(111)

ax.plot(x, y1, label="sin")
ax.plot(x, y2, label="cos")

ax.legend(
    loc="upper left",
    bbox_to_anchor=(1.02, 1),
    borderaxespad=0,
    prop=font_dict
)
font_dict["color"] = "white"
ax.set_title(
    "sin and cos",
    fontdict=font_dict,
    bbox=box_dict
)
fig.suptitle(
    "",
    y=0.95,
    size=16,
    color="gray",
    fontweight="bold"
)
fig.subplots_adjust(left=0.2, bottom=0.2, right=0.75, top=0.8)

plt.show()
